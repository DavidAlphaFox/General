  val exponentiate: t -> int -> t
  val multiply: t -> t -> t
  val negate: t -> t
  val substract: t -> t -> t

  val (**): M.t -> int -> M.t

  val (-): M.t -> M.t -> M.t
  val (/): M.t -> M.t -> M.t
  val (~+): M.t -> M.t
  val (~-): M.t -> M.t
end
