module Examples: sig
  module type Element = sig
    module O: sig
      val (<>): t -> t -> bool
      val (=): t -> t -> bool
    end
    type t
    val different: t -> t -> bool
    val equal: t -> t -> bool
    val repr: t -> bytes
  end
  module type S0 = sig
    type t
    val differences: (t * t) list
    val equalities: t list list
    val representations: (t * bytes) list
    val values: t list
  end
  module type S1 = sig
    module A: Element
    type 'a t
    val differences: (A.t t * A.t t) list
    val equalities: A.t t list list
    val representations: (A.t t * bytes) list
    val values: A.t t list
  end
  module type S2 = sig
    module A: Element
    module B: Element
    type ('a, 'b) t
    val differences: ((A.t, B.t) t * (A.t, B.t) t) list
    val equalities: (A.t, B.t) t list list
    val representations: ((A.t, B.t) t * bytes) list
    val values: (A.t, B.t) t list
  end
  module type S3 = sig
    module A: Element
    module B: Element
    module C: Element
    type ('a, 'b, 'c) t
    val differences: ((A.t, B.t, C.t) t * (A.t, B.t, C.t) t) list
    val equalities: (A.t, B.t, C.t) t list list
    val representations: ((A.t, B.t, C.t) t * bytes) list
    val values: (A.t, B.t, C.t) t list
  end
  module type S4 = sig
    module A: Element
    module B: Element
    module C: Element
    module D: Element
    type ('a, 'b, 'c, 'd) t
    val differences: ((A.t, B.t, C.t, D.t) t * (A.t, B.t, C.t, D.t) t) list
    val equalities: (A.t, B.t, C.t, D.t) t list list
    val representations: ((A.t, B.t, C.t, D.t) t * bytes) list
    val values: (A.t, B.t, C.t, D.t) t list
  end
  module type S5 = sig
    module A: Element
    module B: Element
    module C: Element
    module D: Element
    module E: Element
    type ('a, 'b, 'c, 'd, 'e) t
    val differences: ((A.t, B.t, C.t, D.t, E.t) t * (A.t, B.t, C.t, D.t, E.t) t) list
    val equalities: (A.t, B.t, C.t, D.t, E.t) t list list
    val representations: ((A.t, B.t, C.t, D.t, E.t) t * bytes) list
    val values: (A.t, B.t, C.t, D.t, E.t) t list
  end
end
